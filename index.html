串口调试助手模式组合对照表
接收：ASCII 接收数字解析显示为ASCLL码对应字符 eg:发送0x41 串口助手接收为’A’;
发送：ASCII 发送字符解析并发送为ASCLL码值 eg:串口助手发送’A’ 电脑接收到0x41
接收：Hex接收数字解析显示为ASCLL码值 eg:发送’A’ 串口助手接收为0x41;
发送：Hex发送数字解析并发送为对应ASCLL码对应的字符eg:串口助手发送41 电脑接收到’A’;

scheduler.h

#include “scheduler.h”

typedef struct{
void (task_func*)(void);
uint32_t rate_ms;
uint32_t last_run;
}task_t;

static task_t task_scheduler[]= //static修饰函数全局数组、全局变量时，会将其作用域从所有.c文件,限制为仅当前.c文件可见 //static修饰局部变量时，变量值初始化一次
{

}

static uint8_t task_num;
scheduler_init(){
task_num=sizeof(task_scheduler)/sizeof(task_t);
}

sheduler_run(){
for(uint8_t i=0;i<task_num;i++){
uint32_t now_time=HAL_GetTick();
if(now_time>=task_scheduler[i].rate_ms+task_scheduler[i].last_run){
task_scheduler[i].last_run=uwTick;
task_scheduler[i].task_func();
}
}
}

/*main函数中：
sheduler_init();
*/

led_user.h

#include”led_user.h”

void led_disp(uint8_t *ucled)
{
uint8_t temp= 0x00;
static uint8_t temp_old=0xff;

for (uint8_t i=0; i<8; i++)
{
  temp|=ucled[i]<<(7-i);
}
if(temp!=temp_old)
{
GPIOC->ODR&= 0x00ff;
GPIOC->ODR|= ((~temp)&0x00ff)<< 8;
GPIOD->BSRR= 0x01<< 2;
GPIOD->BRR= 0x01<< 2;
temp_old=temp;
}

}

uint8_t ucled[8]={1,1,1,1,0,0,0,0};

void led_proc(){
led_disp(ucled);
}

/*
main函数中：
system_init();//对GPIOC显示初始化

key_user.h

#include “key_user.h”

uint8_t key_val;
uint8_t key_down;
uint8_t key_up;
uint8_t key_old;

uint8_t key_read()
{
uint8_t temp=0;
if(HAL_GPIO_ReadPin(GPIOB,GPIO_PIN_1)==GPIO_PIN_RESET)
{
temp=1;
}
if(HAL_GPIO_ReadPin(GPIOB,GPIO_PIN_2)==GPIO_PIN_RESET)
{
temp=2;
}
if(HAL_GPIO_ReadPin(GPIOB,GPIO_PIN_3)==GPIO_PIN_RESET)
{
temp=3;
}
if(HAL_GPIO_ReadPin(GPIOB,GPIO_PIN_4)==GPIO_PIN_RESET)
{
temp=4;
}
return temp;
}

void key_proc()
{
key_val=key_read(); //PB2按下瞬间及保持=2，松开=0
key_down=key_val&(key_val^key_old); //2=2&（2^0） 0^x=x;
key_up=(key_val)&(key_val^key_old); //2=0&（0^2）=0xff & 2
key_old=key_val;

if(key_down!=0)
	ucled[key_down-1]^=1;							//对应的led电平翻转

}
