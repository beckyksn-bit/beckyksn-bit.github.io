adc_user.h

#include “adc_user.h”
#include “adc_user.h”
#include “adc_user.h”

uint32_t dma_buff[2][30];
float adc_value[2];

void adc_proc(void)
{
for(int i=0;i<30;i++)
{
adc_value[0]+=dma_buff[0][i];
adc_value[1]+=dma_buff[1][i];
}
adc_value[0]=adc_value[0]/303.3f/4095;
adc_value[1]=adc_value[1]/303.3f/4095;

LcdSprintf(Line2,”adc_value[0]为%.2f”,adc_value[0]);
LcdSprintf(Line3,”adc_value[1]为%.2f”,adc_value[1]);
}
