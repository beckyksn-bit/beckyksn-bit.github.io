led_user.h

#include”led_user.h”

void led_disp(uint8_t *ucled)
{
uint8_t temp= 0x00;
static uint8_t temp_old=0xff;

for (uint8_t i=0; i<8; i++)
{
  temp|=ucled[i]<<(7-i);
}
if(temp!=temp_old)
{
GPIOC->ODR&= 0x00ff;
GPIOC->ODR|= ((~temp)&0x00ff)<< 8;
GPIOD->BSRR= 0x01<< 2;
GPIOD->BRR= 0x01<< 2;
temp_old=temp;
}

}

uint8_t ucled[8]={1,1,1,1,0,0,0,0};

void led_proc(){
led_disp(ucled);
}

main函数中：
system_init();//对GPIOC显示初始化
